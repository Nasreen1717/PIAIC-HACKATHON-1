### 1.2 System Components

| Component | Chapter | Role | Latency Target |
|-----------|---------|------|-----------------|
| **Whisper Transcriber** | 10 | Audio → Text | &lt;2 seconds |
| **LLM Planner** | 11 | Text → Task Plan | &lt;5 seconds |

### 1.3 Information Flow

```
Step 1: Voice Capture
  • Duration: 2 seconds
  • Audio: 16-bit PCM, 16kHz, mono
  • Size: ~64 KB

Step 2: Transcription (Whisper)
  • Input: WAV audio bytes
  • Processing: 1-2 seconds
  • Output: transcript + confidence

Step 3: Task Decomposition (LLM)
  • Input: "bring me red cube"
  • Processing: 3-5 seconds (API latency + thinking)
  • Output: JSON array of steps

Step 4: Plan Validation
  • Check: action_type, parameters, dependencies
  • Time: &lt;100ms
  • Output: validated_plan or errors

Step 5: Sequential Execution
  • For each step:
    - Call appropriate ROS 2 action/service
    - Monitor for success/failure
    - Log outcome
    - Estimate: 2-5 seconds per step

Step 6: Metrics Collection
  • Track all latencies and outcomes
  • Compile ExecutionTrace message
  • Time: &lt;100ms
```

### 2.3 Error Recovery Strategies

| Error Type | Example | Recovery Strategy |
|-----------|---------|-------------------|
| **Navigation Blocked** | Path obstacle | Retry with alternate route; request user intervention |
| **Object Not Found** | Perception fails | Expand search radius; ask user for location hint |
| **Gripper Slip** | Object drops | Retry grasp 3× with varied grip force |
| **Timeout** | Step exceeds 30s | Log timeout, skip step, continue to next |
| **API Unavailable** | Service crashed | Wait 5 seconds, retry; fallback to manual mode |

## 3. Error Recovery & Failure Handling

### 3.1 Retry Logic

```python
class ErrorRecoveryManager:
    """Handle step failures with appropriate recovery."""

    def handle_step_failure(self, step: ExecutionStep, error: str) -> bool:
        """
        Attempt recovery for failed step.

        Returns:
            True if recovery successful, False if unrecoverable
        """
        action_type = step['action_type']

        # ✅ Navigation failures
        if action_type == 'navigate':
            if 'path_blocked' in error:
                # Retry with alternate route
                return self.retry_navigation_alternate_route(step)
            elif 'timeout' in error:
                # Skip and continue
                return False

        # ✅ Perception failures
        elif action_type == 'perceive':
            if 'object_not_found' in error:
                # Expand search area
                return self.expand_perception_search(step)

        # ✅ Manipulation failures
        elif action_type == 'manipulate':
            if 'gripper_slip' in error:
                # Retry with increased grip force
                return self.retry_grasp_increased_force(step)

        return False
```

```python
class ExecutionLogger:
    """Log all execution events for analysis."""

    def __init__(self):
        self.events = []

    def log_step_start(self, step_index: int, action: str):
        self.events.append({
            'type': 'step_start',
            'step_index': step_index,
            'action': action,
            'timestamp': time.time()
        })

    def log_step_success(self, step_index: int, duration: float, outcome: dict):
        self.events.append({
            'type': 'step_success',
            'step_index': step_index,
            'duration': duration,
            'outcome': outcome,
            'timestamp': time.time()
        })

    def log_step_failure(self, step_index: int, error: str, recovery_attempted: bool):
        self.events.append({
            'type': 'step_failure',
            'step_index': step_index,
            'error': error,
            'recovery_attempted': recovery_attempted,
            'timestamp': time.time()
        })

    def save_execution_trace(self, file_path: str):
        """Save execution trace to JSON for analysis."""
        with open(file_path, 'w') as f:
            json.dump(self.events, f, indent=2)
```

```python
class CapstoneMetrics:
    """Collect metrics for capstone evaluation."""

    def __init__(self):
        self.start_time = time.time()
        self.steps_completed = 0
        self.steps_failed = 0
        self.step_latencies = []
        self.safety_incidents = []
        self.log_messages = []

    def record_step_latency(self, step_index: int, latency: float):
        self.step_latencies.append({
            'step': step_index,
            'latency': latency
        })

    def record_safety_incident(self, incident_type: str, description: str):
        self.safety_incidents.append({
            'type': incident_type,
            'description': description,
            'timestamp': time.time()
        })

    def get_summary(self) -> dict:
        total_duration = time.time() - self.start_time
        total_steps = self.steps_completed + self.steps_failed

        return {
            'total_duration': total_duration,
            'steps_completed': self.steps_completed,
            'steps_failed': self.steps_failed,
            'success_rate': self.steps_completed / max(1, total_steps),
            'avg_step_latency': mean(l['latency'] for l in self.step_latencies),
            'safety_incidents': len(self.safety_incidents),
            'incidents': self.safety_incidents
        }
```

### Example 7: Single-Step Execution with Action Clients

```python
"""
🚀 Module 4 Example 7: Single-Step Execution
Demonstrates executing a single action (navigation) from a task plan.
"""

import rclpy
import time
from rclpy.action import ActionClient
from rclpy.node import Node
from nav2_msgs.action import NavigateToPose
from geometry_msgs.msg import PoseStamped

class SingleStepExecutor(Node):
    """Execute a single navigation step."""

    def __init__(self):
        super().__init__('single_step_executor')
        self.nav_client = ActionClient(self, NavigateToPose, 'navigate_to_pose')

    def execute_navigation_step(self, destination: str) -> dict:
        """
        Execute a single navigation step.

        Args:
            destination: Target location name

        Returns:
            {'status': 'success'|'failure', 'duration': float, 'error': str or None}
        """
        # ✅ Define known locations (map to 3D coordinates)
        locations = {
            'kitchen': (3.0, 2.0, 0.0),
            'bedroom': (5.0, 5.0, 0.0),
            'living_room': (1.0, 4.0, 0.0)
        }

        if destination not in locations:
            return {
                'status': 'failure',
                'duration': 0.0,
                'error': f"Unknown location: {destination}"
            }

        # ✅ Create pose goal
        goal = NavigateToPose.Goal()
        goal.pose.header.frame_id = 'map'
        goal.pose.pose.position.x = float(locations[destination][0])
        goal.pose.pose.position.y = float(locations[destination][1])
        goal.pose.pose.position.z = float(locations[destination][2])
        goal.pose.pose.orientation.w = 1.0

        # ✅ Send goal and wait for result
        start_time = time.time()

        future = self.nav_client.send_goal_async(goal)
        rclpy.spin_until_future_complete(self, future)

        goal_handle = future.result()
        if not goal_handle.accepted:
            return {
                'status': 'failure',
                'duration': time.time() - start_time,
                'error': 'Navigation goal rejected'
            }

        # ✅ Wait for result
        result_future = goal_handle.get_result_async()
        rclpy.spin_until_future_complete(self, result_future)

        result = result_future.result()
        duration = time.time() - start_time

        if result is None:
            return {
                'status': 'failure',
                'duration': duration,
                'error': 'Navigation failed (result is None)'
            }

        return {
            'status': 'success',
            'duration': duration,
            'error': None
        }


def main():
    rclpy.init()
    executor = SingleStepExecutor()

    # ✅ Execute navigation to kitchen
    result = executor.execute_navigation_step('kitchen')

    print(f"Status: {result['status']}")
    print(f"Duration: {result['duration']:.2f}s")
    if result['error']:
        print(f"Error: {result['error']}")

    executor.destroy_node()
    rclpy.shutdown()


if __name__ == '__main__':
    main()
```

### Example 8: Full Multi-Step Capstone Execution with Metrics

```python
"""
🚀 Module 4 Example 8: Full Capstone Execution Pipeline
Demonstrates end-to-end execution from voice input to task completion.
Includes metrics collection and logging.
"""

import rclpy
import time
import json
from dataclasses import dataclass, asdict
from typing import List
from rclpy.node import Node

@dataclass
class StepMetric:
    """Metrics for a single execution step."""
    step_index: int
    action_type: str
    duration: float
    success: bool
    error: str = None

class CapstoneExecutor(Node):
    """Execute full VLA pipeline from voice to completion."""

    def __init__(self):
        super().__init__('capstone_executor')
        self.metrics: List[StepMetric] = []
        self.start_time = None

    def execute_capstone_scenario(self, goal: str) -> dict:
        """
        Execute full capstone scenario: voice → plan → execute → metrics.

        Args:
            goal: High-level goal description

        Returns:
            Execution summary with metrics
        """
        self.start_time = time.time()
        self.metrics = []

        try:
            # ✅ Step 1: Transcription (simulated)
            self.get_logger().info(f"🎤 Goal: '{goal}'")

            # ✅ Step 2: LLM Planning (simulated)
            plan = self.simulate_llm_planning(goal)
            self.get_logger().info(f"📋 Generated plan with {len(plan)} steps")

            # ✅ Step 3: Execute each step
            for step_index, step in enumerate(plan):
                success, duration, error = self.execute_step(step_index, step)
                self.metrics.append(
                    StepMetric(
                        step_index=step_index,
                        action_type=step['action'],
                        duration=duration,
                        success=success,
                        error=error
                    )
                )

                status = "✅" if success else "❌"
                self.get_logger().info(
                    f"{status} Step {step_index}: {step['action']} ({duration:.2f}s)"
                )

                if not success:
                    self.get_logger().warn(f"   Error: {error}")

            # ✅ Step 4: Compile results
            total_duration = time.time() - self.start_time
            success_count = sum(1 for m in self.metrics if m.success)

            summary = {
                'goal': goal,
                'total_duration': total_duration,
                'steps_completed': success_count,
                'steps_failed': len(self.metrics) - success_count,
                'success_rate': success_count / max(1, len(self.metrics)),
                'metrics': [asdict(m) for m in self.metrics]
            }

            return summary

        except Exception as e:
            self.get_logger().error(f"❌ Capstone execution failed: {str(e)}")
            raise

    def execute_step(self, step_index: int, step: dict) -> tuple:
        """
        ✅ Execute a single step in the plan.

        Returns:
            (success, duration, error_message)
        """
        start_time = time.time()
        action = step['action']

        try:
            if action == 'navigate':
                # Simulated navigation (2-5 seconds)
                time.sleep(3.0)
                return True, time.time() - start_time, None

            elif action == 'perceive':
                # Simulated perception (1-2 seconds)
                time.sleep(1.5)
                return True, time.time() - start_time, None

            elif action == 'manipulate':
                # Simulated manipulation (2-3 seconds)
                time.sleep(2.5)
                return True, time.time() - start_time, None

            else:
                return False, time.time() - start_time, f"Unknown action: {action}"

        except Exception as e:
            return False, time.time() - start_time, str(e)

    def simulate_llm_planning(self, goal: str) -> List[dict]:
        """Simulate LLM planning (return predefined plans for demo)."""
        # Simple plans for demo purposes
        plans = {
            'move': [
                {'action': 'navigate', 'destination': 'living_room'},
            ],
            'pick': [
                {'action': 'navigate', 'destination': 'kitchen'},
                {'action': 'perceive', 'target': 'red cube'},
                {'action': 'manipulate', 'action': 'grasp'}
            ],
            'bring': [
                {'action': 'navigate', 'destination': 'kitchen'},
                {'action': 'perceive', 'target': 'red cube'},
                {'action': 'manipulate', 'action': 'grasp'},
                {'action': 'navigate', 'destination': 'living_room'},
                {'action': 'manipulate', 'action': 'release'}
            ]
        }

        # Match goal to plan
        for keyword, plan in plans.items():
            if keyword in goal.lower():
                return plan

        # Default plan
        return [
            {'action': 'navigate', 'destination': 'user_location'},
            {'action': 'perceive', 'target': 'any'},
            {'action': 'manipulate', 'action': 'grasp'}
        ]


def main():
    rclpy.init()
    executor = CapstoneExecutor()

    # ✅ Run capstone scenarios
    scenarios = [
        "Move the red cube",
        "Bring me the blue bottle from the kitchen",
        "Fetch the object on the table and bring it to the bedroom"
    ]

    all_results = []
    for scenario in scenarios:
        print(f"\n{'='*60}")
        print(f"Executing: {scenario}")
        print('='*60)

        result = executor.execute_capstone_scenario(scenario)
        all_results.append(result)

        print(f"\n📊 Summary:")
        print(f"  Total Duration: {result['total_duration']:.2f}s")
        print(f"  Success Rate: {result['success_rate']:.0%}")
        print(f"  Steps: {result['steps_completed']}/{result['steps_completed'] + result['steps_failed']}")

    # ✅ Save all results to JSON
    with open('/tmp/capstone_results.json', 'w') as f:
        json.dump(all_results, f, indent=2)

    print(f"\n✅ Results saved to /tmp/capstone_results.json")

    executor.destroy_node()
    rclpy.shutdown()


if __name__ == '__main__':
    main()
```
